Code Example

// GameScript3 IDE - Full project example

import "GameScript3"
import "GameScript3.Graphics"
import "GameScript3.Physics"

class Player extends Actor
    health = 100
    speed = 15
    
    function onStart()
        print("Player initialized")
        this.setCollider("box", 32, 32)
    end
    
    function update(deltaTime)
        handleInput()
        updatePhysics(deltaTime)
    end
    
    function handleInput()
        if Input.isKeyPressed("W") then
            this.move(0, -speed)
        end
        if Input.isKeyPressed("A") then
            this.move(-speed, 0)
        end
    end
    
    function takeDamage(damage)
        health = health - damage
        if health <= 0 then
            this.destroy()
        end
    end
end

game = new Game()
game.run()
