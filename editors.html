Quick Start Example

// Create a simple game project in either editor

import "GameScript3"

class MyGame extends Game
    function onStart()
        print("Game Started!")
        createPlayer()
    end
    
    function createPlayer()
        player = new Player()
        player.setPosition(500, 300)
        player.setHealth(100)
    end
end

game = new MyGame()
game.run()
