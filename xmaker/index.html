Quick Start Example

// XMaker - Simple game example

import "GameScript3"

class SimpleGame extends Game
    function onStart()
        print("Welcome to XMaker!")
        createScene()
    end
    
    function createScene()
        player = new Sprite()
        player.load("player.png")
        player.setPosition(400, 300)
    end
end

// Run the game
game = new SimpleGame()
game.run()
